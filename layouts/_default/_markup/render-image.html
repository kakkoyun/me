{{- $dest := .Destination -}} {{- $alt := .Text -}} {{- $title := .Title -}} {{- $loading := "lazy" -}} {{- $fetchpriority := "" -}} {{- if .Page.IsHome -}} {{- $loading = "eager" -}} {{- $fetchpriority = "high" -}} {{- end -}} {{- $img := .Page.Resources.GetMatch $dest -}} {{/* Fall back to a global asset for absolute paths (e.g. /img/banner.jpeg). */}} {{- if not $img -}} {{- $assetPath := strings.TrimPrefix "/" $dest -}} {{- if and (hasPrefix $dest "/") (not (or (hasPrefix $dest "//") (hasPrefix $dest "http"))) -}} {{- $img = resources.Get $assetPath -}} {{- end -}} {{- end -}} {{- if and $img $img.MediaType (in (slice "image/jpeg" "image/png" "image/gif" "image/bmp" "image/tiff" "image/webp") $img.MediaType.Type) -}} {{- $sizes := slice 480 1024 1920 -}} {{- $srcset := slice -}} {{- $srcsetWebP := slice -}} {{- range $sizes -}} {{- if le . $img.Width -}} {{- $resized := $img.Resize (printf "%dx" .) -}} {{- $srcset = $srcset | append (printf "%s %dw" $resized.RelPermalink .) -}} {{- $resizedW := $img.Resize (printf "%dx webp" .) -}} {{- $srcsetWebP = $srcsetWebP | append (printf "%s %dw" $resizedW.RelPermalink .) -}} {{- end -}} {{- end -}} {{- with $srcsetWebP }}{{ end }} {{- with $srcset }}{{ end }} {{- else -}} {{- end -}}
